JavaScript


            var editors = [].slice.call(document.querySelectorAll('.editor'));
            editors.forEach(function (div) {
                div.addEventListener('click', function (e) {
                    if (!Zodit.instances[div.id]) {
                        Object.keys(Zodit.instances).forEach(function (id) {
                            Zodit.instances[id].destruct();
                        });

                        var editor = new Zodit(div, {
                            preset: 'inline'
                        });

                        editor.s.insertCursorAtPoint(e.clientX, e.clientY);
                    }
                });
            });
